93 int main(void)
94 {
95  unsigned t = time(NULL);
96  unsigned l[VMAX], lt[VMAX][VMAX], tl[VMAX][VMAX], qt, ql, chosen[VMAX], v[VMAX];
97  unsigned i, j, k, m, flag, it_escape, lctk, alt, atl;
98  srand(time(NULL));
99  vzeros(l, VMAX);
100  vzeros(v, VMAX);
101  mzeros(lt, VMAX, VMAX);
102  mzeros(tl, VMAX, VMAX);
103  scanf("%u",&ql);
104  scanf("%u",&qt);
105  if(DEBUG>1) printf("Quantidade de lugares: %u\n",ql);
106  if(ql>VMAX)
107  {
108  printf("Erro linha 1: quantidade de lugares acima do maximo. \n");
109  return -1;
110  }
111  if(DEBUG>1) printf("Quantidade de transicoes: %u\n",qt);
112  if(qt>VMAX)
113  {
114  printf("Erro linha 2: quantidade de transicoes acima do maximo. \n");
115  return -1;
116  }
117  scanf("%u",&lctk);
118  scanf("%u",&alt);
119  scanf("%u",&atl);
120  for(k=0;k<lctk;k++)
121  {
122  scanf("%u %u", &i, &j);
123  if(i>=ql)
124  {
125  printf("Erro linha %u: lugar maior que o definido. \n", 6+k);
126  return -1;
127  }
128  l[i]=j;
129  }
130  for(k=0;k<alt;k++)
131  {
132  scanf("%u %u %u", &i, &lctk, &j);
133  if(i>=ql||j>=qt)
134  {
135  printf("Erro linha %u: lugar ou transicao acima do definido. \n", 6+lctk+k);
136  return -1;
137  }
138  lt[i][j]= lctk;
139  }
140  for(k=0;k<atl;k++)
141  {
142  scanf("%u %u %u", &i, &lctk, &j);
143  if(i>=qt||j>=ql)
144  {
145  printf("Erro linha %u: lugar ou transicao acima do definido. \n", 6+lctk+k+alt);
146  return -1;
147  }
148  tl[i][j] = lctk;
149  }
150  if(DEBUG>1) printf("Token em cada lugar:[");
151  if(DEBUG>1) printv(l,ql);
152  if(DEBUG>1) printf(" ]\n");
153  if(DEBUG>1) printf("Matriz lt:");
154  if(DEBUG>1) printm(lt,ql,qt);
155  if(DEBUG>1) printf("\nMatriz tl:");
156  if(DEBUG>1) printm(tl,qt,ql);
157  if(DEBUG>1) printf("\n");
158  printf("======= INICIO DA SIMULACAO =======\n");
159  for(k=0;k<ITER;k++)
160  {
161  if(DEBUG>1) printf("\nInteracao %u:\n",k+1);
162  rlist(chosen,qt);
163  it_escape = 0;
164  if(DEBUG>1) printf("Ordem de tentativa:");
165  if(DEBUG>1) printv(chosen,qt);
166  if(DEBUG>1) printf("\n");
167  for(m=0;m<qt;m++)
168  {
169  i = chosen[m]; //sorteio i
170  if(DEBUG>0) printf("Tentativa %u: numero sorteado: [%u]\n",m+1,i);
171  flag = 1;
172  for(j=0;j<ql;j++)
173  if(lt[j][i] != 0)
174  if(l[j] < lt[j][i])
175  {
176  flag = 0;
177  break;
178  }
179  if(flag)
180  {
181  it_escape = 1; //alguma transicao ativou
182  if((rand()%100 + 1) <= PCT)
183  {
184  if(DEBUG>0) printf("Transicao %u ativou!\n",i);
185  for(j=0;j<ql;j++)
186  if(lt[j][i] != 0)
187  {
188  l[j] -= lt[j][i];
189  if(DEBUG>1) printf("Transicao %u consumiu %u token(s) do lugar %u \n", i, lt[j][i], j);
190  }
191  for(j=0;j<ql;j++)
192  if(tl[i][j] != 0)
193  {
194  l[j] += tl[i][j];
195  v[i]++;
196  if(DEBUG>1) printf("Transicao %u produziu %u token(s) \n", i, tl[i][j]);
197  }
198  if(DEBUG>0) printf("Token em cada lugar:[");
199  if(DEBUG>0) printv(l,ql);
200  if(DEBUG>0) printf(" ]\n");
201  }
202  else
203  if(DEBUG>0) printf("Transicao %u decidiu nao ativar.\n",i);
204  }
205  }
206  if(it_escape == 0) //se nada aconteceu com nenhuma transicao
207  break;
208  }
209  printf("======= FIM DA SIMULACAO ==========\n");
210  t = time(NULL) - t;
211  printf("Tempo de reprodução do programa: %u segundo(s).\n",t);
212  if(k==0)
213  printf("Aviso: nenhuma transicao foi disparada! \n");
214  printf("Numero de iteracoes: %u.\n",k);
215  printf("Lugares com token: \n");
216  for(j=0;j<ql;j++)
217  if(l[j] != 0)
218  printf("Lugar %u: %u tokens \n",j,l[j]);
219 
220  printf("Transicoes disparadas: \n");
221  for(j=0;j<ql;j++)
222  if(v[j] != 0)
223  printf("Transicao %u: disparada %u vezes \n",j,v[j]);
224 
225 
226  return EXIT_SUCCESS;
227 }
